I originally thought that the best method to do this might be to use the join method and split method however I quickly realized why this might not work. Because it will fail whenever a string conntains a space such as "Hello World". This is because split will split every single string into single words in a list no matter what.

```python
class Solution:

    def encode(self, strs: List[str]) -> str:
        encoded = " ".join(strs)
        return encoded

    def decode(self, s: str) -> List[str]:
        decoded = s.split()
        return decoded
```

> Time Complexity : `O()`
> 
> Space Complexity : `O()`

'''python
class Solution:

    def encode(self, strs: List[str]) -> str:
        result = ""
        for string in strs:
            result += str(len(string)) + "#" + string
        return result


    def decode(self, s: str) -> List[str]:
        result, index = [], 0

        while index < len(s):
            j = index
            while s[j] != "#":
                j += 1
            length = int(s[index:j])
            result.append(s[j + 1 : j + 1 + length])
            index = j + 1 + length
        return result
